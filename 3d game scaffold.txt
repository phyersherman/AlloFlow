        // --- AUDIO SYSTEM ---
        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        let soundEnabled = false; 

        function playSound(type) {
            if (!soundEnabled || audioCtx.state === 'suspended') {
                if(audioCtx.state === 'suspended') audioCtx.resume();
                soundEnabled = true;
            }
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain);
            gain.connect(audioCtx.destination);

            const now = audioCtx.currentTime;
            
            if (type === 'step') {
                osc.type = 'triangle';
                osc.frequency.setValueAtTime(80, now);
                osc.frequency.exponentialRampToValueAtTime(10, now + 0.08);
                gain.gain.setValueAtTime(0.05, now);
                gain.gain.exponentialRampToValueAtTime(0.001, now + 0.08);
                osc.start(now);
                osc.stop(now + 0.08);
            } else if (type === 'interact') {
                osc.type = 'sine';
                osc.frequency.setValueAtTime(600, now);
                osc.frequency.linearRampToValueAtTime(800, now + 0.1);
                gain.gain.setValueAtTime(0.05, now);
                gain.gain.linearRampToValueAtTime(0, now + 0.2);
                osc.start(now);
                osc.stop(now + 0.2);
            } else if (type === 'success') {
                [523.25, 659.25, 783.99].forEach((freq, i) => { // C Major
                    const o = audioCtx.createOscillator();
                    const g = audioCtx.createGain();
                    o.connect(g);
                    g.connect(audioCtx.destination);
                    o.type = 'triangle';
                    o.frequency.value = freq;
                    g.gain.setValueAtTime(0.05, now + i*0.05);
                    g.gain.linearRampToValueAtTime(0, now + i*0.05 + 0.4);
                    o.start(now + i*0.05);
                    o.stop(now + i*0.05 + 0.4);
                });
            } else if (type === 'tension') {
                osc.type = 'sawtooth';
                osc.frequency.setValueAtTime(100, now);
                osc.frequency.linearRampToValueAtTime(50, now + 0.5);
                gain.gain.setValueAtTime(0.05, now);
                gain.gain.linearRampToValueAtTime(0, now + 0.5);
                osc.start(now);
                osc.stop(now + 0.5);
            } else if (type === 'bell') {
                osc.type = 'square';
                osc.frequency.setValueAtTime(660, now);
                const lfo = audioCtx.createOscillator();
                lfo.frequency.value = 15;
                const lfoGain = audioCtx.createGain();
                lfoGain.gain.value = 10;
                lfo.connect(lfoGain);
                lfoGain.connect(osc.frequency);
                lfo.start(now); lfo.stop(now + 1.5);
                gain.gain.setValueAtTime(0.1, now);
                gain.gain.linearRampToValueAtTime(0, now + 1.5);
                osc.start(now);
                osc.stop(now + 1.5);
            } else if (type === 'paper') {
                osc.type = 'sine';
                osc.frequency.setValueAtTime(800, now);
                gain.gain.setValueAtTime(0.02, now);
                gain.gain.linearRampToValueAtTime(0, now + 0.1);
                osc.start(now); osc.stop(now+0.1);
            }
        }

        function playBlip() {
            if (!soundEnabled) return;
            const now = audioCtx.currentTime;
            const osc = audioCtx.createOscillator();
            const g = audioCtx.createGain();
            osc.type = 'sine';
            osc.frequency.setValueAtTime(400 + Math.random() * 200, now);
            g.gain.setValueAtTime(0.02, now);
            g.gain.exponentialRampToValueAtTime(0.001, now + 0.05);
            osc.connect(g); g.connect(audioCtx.destination);
            osc.start(); osc.stop(now + 0.05);
        }

        async function fetchGemini(prompt) {
            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },

                });

                return data.candidates[0].content.parts[0].text;

        function updateMovement() {
            if (gameState.inDialogue) return;
            
            let prompt = "";
            let nearBench = false;
            gameState.benchList.forEach(b => {
                if(Math.hypot(camera.position.x - b.x, camera.position.z - b.z) < 3.2) {
                    nearBench = true;
                    prompt = gameState.isSitting ? "Press E to Stand Up" : "Press E to Rest";
                    if(keys.e) { 
                        gameState.isSitting = !gameState.isSitting; 
                        keys.e = false; 
                        playSound('interact'); 
                    }
                }
            });

            if(gameState.isSitting) {
                gameState.stamina = Math.min(100, gameState.stamina + 1.2);
            } else {
                const isMoving = keys.w || keys.s || keys.a || keys.d;
                let speed = keys.shift && gameState.stamina > 0 ? 0.5 : 0.25;
                if(isMoving && keys.shift) gameState.stamina = Math.max(0, gameState.stamina - 0.5); 
                else gameState.stamina = Math.min(100, gameState.stamina + 0.2);

                camera.rotation.order = 'YXZ'; camera.rotation.y = yaw; camera.rotation.x = pitch;
                const fwd = new THREE.Vector3(0, 0, -1).applyAxisAngle(new THREE.Vector3(0, 1, 0), yaw);
                const rgt = new THREE.Vector3(1, 0, 0).applyAxisAngle(new THREE.Vector3(0, 1, 0), yaw);
                let dx = 0, dz = 0;
                if (keys.w) { dx += fwd.x * speed; dz += fwd.z * speed; }
                if (keys.s) { dx -= fwd.x * speed; dz -= fwd.z * speed; }
                if (keys.a) { dx -= rgt.x * speed; dz -= rgt.z * speed; }
                if (keys.d) { dx += rgt.x * speed; dz += rgt.z * speed; }
                if(!checkCollision(camera.position.x + dx, camera.position.z, 0.8)) camera.position.x += dx;
                if(!checkCollision(camera.position.x, camera.position.z + dz, 0.8)) camera.position.z += dz;
            }

            document.getElementById('stamina-bar').style.width = gameState.stamina + '%';

            if(!prompt) {
                zones.forEach(z => { if(Math.hypot(camera.position.x - z.position.x, camera.position.z - z.position.z) < 3.5) { prompt = "Press E to Mediate"; if(keys.e) startScene(z.userData.scenarioId); } });
                notes.forEach(n => { if(!n.data.collected && Math.hypot(camera.position.x - n.data.x, camera.position.z - n.data.z) < 2.5) { prompt = "Press E to Read Note"; if(keys.e) { n.data.collected = true; scene.remove(n.mesh); gameState.score += 50; document.getElementById('score-display').innerText = gameState.score; playSound('paper'); } } });
            }
            document.getElementById('interaction-prompt').style.display = prompt ? 'block' : 'none';
            document.getElementById('interaction-prompt').innerText = prompt;
        }

        function spawnSplash(x, z) {
            const geo = new THREE.RingGeometry(0.1, 0.2, 12);
            const mat = new THREE.MeshBasicMaterial({color: 0x81d4fa, transparent: true, opacity: 0.6, side: THREE.DoubleSide});
            const m = new THREE.Mesh(geo, mat);
            m.rotation.x = -Math.PI/2; m.position.set(x, 0.51, z);
            scene.add(m); splashMeshes.push({mesh: m, age: 0});
        }

        function animate() {
            requestAnimationFrame(animate); updateMovement(); drawMinimap();
            
            // Environmental Animations
            fountainParticles.forEach(fp => {
                const p = fp.system.geometry.attributes.position.array;
                for(let i=0; i<fp.velocities.length; i++) {
                    p[i*3] += fp.velocities[i].x; p[i*3+1] += fp.velocities[i].y; p[i*3+2] += fp.velocities[i].z;
                    fp.velocities[i].y -= 0.005;
                    if(p[i*3+1] < 0.5) { 
                        if (Math.random() > 0.9) spawnSplash(p[i*3], p[i*3+2]);
                        p[i*3] = fp.origin.x; p[i*3+1] = fp.origin.y; p[i*3+2] = fp.origin.z; fp.velocities[i].y = Math.random()*0.1+0.1; 
                    }
                }
                fp.system.geometry.attributes.position.needsUpdate = true;
            });

            for(let i=splashMeshes.length-1; i>=0; i--) {
                const s = splashMeshes[i]; s.age++;
                s.mesh.scale.multiplyScalar(1.05);
                s.mesh.material.opacity *= 0.95;
                if(s.age > 20) { scene.remove(s.mesh); splashMeshes.splice(i,1); }
            }

            butterflyParticles.forEach(bp => {
                bp.phase += 0.05; bp.mesh.position.y = 1 + Math.sin(bp.phase)*0.5;
                bp.mesh.position.x += Math.cos(bp.phase*0.5)*0.02; bp.mesh.rotation.y += 0.1;
            });

            // UI & Effects
            beacons.forEach(b => { b.rotation.y += 0.02; b.position.y = 3.5 + Math.sin(Date.now()*0.003)*0.2; });
            labelElements.forEach(le => {
                const s = le.pos.clone().project(camera);
                if(s.z < 1) { le.el.style.left = `${(s.x*.5+.5)*window.innerWidth}px`; le.el.style.top = `${(s.y*-.5+.5)*window.innerHeight}px`; le.el.classList.add('visible'); }
                else le.el.classList.remove('visible');
            });

            // NPC Animations
            characters.forEach(c => {
                if (c.userData.talkTimer > 0) {
                    c.userData.talkTimer--;
                    c.children[1].rotation.x = Math.sin(Date.now() * 0.02) * 0.1;
                    c.userData.lHand.position.y = 1.0 + Math.sin(Date.now() * 0.015) * 0.3;
                    c.userData.rHand.position.y = 1.0 + Math.cos(Date.now() * 0.015) * 0.3;
                    if (c.userData.talkTimer % 8 === 0) playBlip();
                }
            });

            // Camera Shake & Leroy
            if(gameState.shakeIntensity > 0) {
                camera.position.x += (Math.random()-0.5) * gameState.shakeIntensity;
                camera.position.y += (Math.random()-0.5) * gameState.shakeIntensity;
                gameState.shakeIntensity *= 0.92;
            }

            if(gameState.inDialogue && gameState.cameraTarget) { camera.position.lerp(gameState.cameraTarget, 0.05); }
            renderer.render(scene, camera);
        }

        async function sendMediation() {
            const i = document.getElementById('user-input'); const v = i.value.trim(); if(!v || i.disabled) return;
            i.value = ''; i.disabled = true; addMsg(v, 'user'); gameState.conversationHistory.push(`Mediator: ${v}`);
            
            const p = `High School Mediation Scenario: ${gameState.currentScenario.intro}\nHistory: ${gameState.conversationHistory.join('\n')}\nRespond as student. Format: STATUS: [ESCALATED/RESOLVED/NEUTRAL], TENSION: [0-100], RESPONSE: [Text]`;
            
            try {
                const r = await fetchGemini(p);
                const tension = parseInt(r.match(/TENSION:\s*(\d+)/i)?.[1] || 50);
                const resp = r.match(/RESPONSE:\s*([\s\S]+)/i)?.[1] || r;
                
                // Tension Spike Logic
                if (tension - gameState.lastTension > 15) {
                    gameState.shakeIntensity = 0.4;
                    document.getElementById('vignette').classList.add('tension-spike');
                    setTimeout(() => document.getElementById('vignette').classList.remove('tension-spike'), 400);
                    playSound('tension');
                }
                gameState.lastTension = tension;

                document.getElementById('tension-fill').style.width = tension + '%';
                addMsg(`STUDENT: "${resp}"`, 'npc'); 
                
                // Trigger animation on relevant NPCs
                characters.forEach(c => {
                    if (c.userData.id === gameState.currentScenario.id) c.userData.talkTimer = resp.length * 0.8;
                });

                i.disabled = false; i.focus();
                if(r.includes('RESOLVED')) { triggerConfetti(camera.position); playSound('success'); }
            } catch (e) {
                addMsg("SYSTEM: Feedback error. Connection shaky.", "system");
                i.disabled = false;
            }
        }

        function playSound(type) {
            if (!soundEnabled) return;
            const now = audioCtx.currentTime;
            const osc = audioCtx.createOscillator();
            const g = audioCtx.createGain();
            osc.connect(g); g.connect(audioCtx.destination);
            
            if (type === 'interact') {
                osc.type = 'sine'; osc.frequency.setValueAtTime(600, now); osc.frequency.linearRampToValueAtTime(800, now + 0.1);
                g.gain.setValueAtTime(0.05, now); g.gain.linearRampToValueAtTime(0, now + 0.2);
                osc.start(); osc.stop(now + 0.2);
            } else if (type === 'success') {
                [523.25, 659.25, 783.99].forEach((freq, i) => {
                    const o = audioCtx.createOscillator(); const gain = audioCtx.createGain();
                    o.connect(gain); gain.connect(audioCtx.destination);
                    o.type = 'triangle'; o.frequency.value = freq;
                    gain.gain.setValueAtTime(0.05, now + i*0.05); gain.gain.linearRampToValueAtTime(0, now + i*0.05 + 0.4);
                    o.start(now + i*0.05); o.stop(now + i*0.05 + 0.4);
                });
            } else if (type === 'paper') {
                osc.type = 'sine'; osc.frequency.setValueAtTime(800, now);
                g.gain.setValueAtTime(0.02, now); g.gain.linearRampToValueAtTime(0, now + 0.1);
                osc.start(); osc.stop(now+0.1);
            } else if (type === 'tension') {
                osc.type = 'sawtooth'; osc.frequency.setValueAtTime(100, now); osc.frequency.linearRampToValueAtTime(50, now + 0.5);
                g.gain.setValueAtTime(0.05, now); g.gain.linearRampToValueAtTime(0, now + 0.5);
                osc.start(); osc.stop(now + 0.5);
            }
        }
